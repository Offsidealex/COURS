\fcolorbox{black}{white}{\textbf{\ding{202} Chute libre - voir TP2}}\\
\vspace{-3mm}
\begin{itemize*}
\item La chute libre est le mouvement dans le \textbf{vide} d'un solide soumis \textbf{uniquement à l'attraction terrestre. }
\vfill 


\ctext{
Remarque : lorsque les forces de résistance de l'air et de la poussée d'Archimède sont négligeables, on admet qu'un solide en tombant dans l'atmosphère est en chute libre. 
}
\vfill 
\begin{minipage}{.54\textwidth}%
 \item

 \ctext{ Un solide en chute libre, lâché sans vitesse initiale, est en \textbf{mouvement rectiligne} et son accélération est \textbf{constante}. \\
        \item - son déplacement est en ligne droite et \textbf{vertical}. \\
       \item - dans le cas de la chute libre, l'accélération $a$ (en $\mathrm{m/s^2}$ est égale à $g$ l'accélération de la pesanteur 
} 

 \item \ctext{La vitesse instantanée $v$ du solide augment proportionnellement au temps : \\
 $$v=g \times t = 9,8 \times t$$}

\end{minipage}%
\hfill
\begin{minipage}{.40\textwidth}%
\includegraphics[width=\textwidth]{Chapitre 2 - Accélération et vitesse dans le cas d’un mouvement rectiligne/images/2 - Cours1.png}

\end{minipage}%

\vfill 

\item 
\noindent%
\begin{minipage}{.54\textwidth}%
 \ctext{La chute libre est un mouvement uniformément accéléré : à chaque seconde de chute libre, la vitesse du solide augmente de 9,8 m/s. L'accélération est donc de 9,8 m/s par seconde donc 9,8 $m/s^2$
 \vspace{1cm}
 Remarque : La vitesse en chute libre est indépendante de la masse du solide. 
 \vspace{1cm} }

\end{minipage}%
\hfill
\begin{minipage}{.40\textwidth}%
\includegraphics[width=\textwidth]{Chapitre 2 - Accélération et vitesse dans le cas d’un mouvement rectiligne/images/2 - Cours2.png}
\end{minipage}%
\end{itemize*}

\fcolorbox{black}{white}{\textbf{\ding{203} Mouvement rectiligne d'accélération constante}}\\
\vspace{-3mm}
\begin{itemize*}

\item Mouvement rectiligne uniformément varié [MRUV]

\noindent%
\begin{minipage}{.35\textwidth}%
 \ctext{Lorsqu'un solide est en mouvement rectiligne uniformément varié : \\
 \begin{itemize}
     \item son déplacement se fait en ligne droite
     \item son accélération $a$ en $m/s^2$ est constante
     \item si a>0 alors la vitesse augmente. 
     \item si a<0 alors la vitesse diminue
     \item la vitesse initiale $v_0$ est la vitesse de départ à l'instant $t=0 s$
     \vspace{1cm}
     \item La relation entre vitesse et accélération est : \\
     \Large{$$v=a \times t + v_0$$}
 \end{itemize}}

\end{minipage}%
\hfill
\begin{minipage}{.55\textwidth}%
\fbox{\includegraphics[width=\textwidth]{Chapitre 2 - Accélération et vitesse dans le cas d’un mouvement rectiligne/images/2 - Cours3.png}} \\[5mm]

\fbox{\includegraphics[width=\textwidth]{Chapitre 2 - Accélération et vitesse dans le cas d’un mouvement rectiligne/images/2 - Cours4.png}}
\end{minipage}%

\vfill 

\item Mouvement rectiligne uniforme[MRU] (Voir TP1) \\ 
\begin{fleche}
\item L'accélération est nulle : $a=0 m/s^2$
\item La vitesse est constante : $v=v_O=constante$
\end{fleche}
