\fcolorbox{black}{blue!30}{\textbf{Calculs préparatoires}}
\begin{itemize*}
    \item Calculez la masse m, en g, de soude nécessaire. 
    \ctext{Calcul de la masse molaire M(NaOH) = M(Na) +M(O)+M(H) = 23 + 16 + 1 = 40 g/mol \\[2mm]
    On veut prépare V = 100 mL soit V = 0,1 L de solution. \\[2mm]
    Il nous faut alors n = $C \times V = 0,2 \times 0,1 = 0,02 mol$ \\[2mm]
    On doit donc peser $m= n \times M = 0,02 \times 40 = 0,8 g$
    }

        \item  Les balances du laboratoire sont précises à 0,1 g près, peut-on les utiliser ? justifier votre réponse. 
    \ctext{La balance n'a pas une précision assez importante pour peser 0,8g \\[2mm] 
    En effet avec une précision de 0,1, on aura 0,1/0,8 = 12,5 \% d'erreur possible. \\[2mm]
    Nore masse masse serait comprise entre 0,7 g < m < 0,9 g}

    \item Calculer, en mol/L, la concentration molaire \ch{C_0} de cette solution. 

    \ctext{On a pesé 4g, on a donc $n=\dfrac{4}{40} = 0,1 mol$ d'hydroxyde de sodium NaOH. \\[2mm]
    On les a dissout dans V= 100 mL d'eau soit V = 0,1 L.\\[2mm]
    on a alors une concentration $C_0=\dfrac{n}{V}=\dfrac{0,1}{0,1} = 1 mol/L$} 
\end{itemize}

\fcolorbox{black}{blue!30}{\textbf{Protocole dilution}} \\[3mm]

\item Pour réaliser une dilution on utilisera le protocole ci-dessous : 
\begin{center}
    \includegraphics[scale=0.6]{Chapitre 4 - Solutions aqueuses/images/dilution2.png}
\end{center}


\item Réaliser la dilution , on prélèvera 20 mL de solution mère en deux fois.

\item Calculer, en mol/l, la concentration \ch{C} en soude de cette solution diluée. 

\ctext{
Nous avez prélevé 20 mL que nous avons dilué dans 100 mL ; $\dfrac{100}{5}=5$ \\[2mm]
Nous avons dilué 5 fois \\[2mm]
la concentration est donc divisée par 5. $C = \dfrac{C_0}{5} = \dfrac{1}{5} = 0,2 mol/L$ 
}
