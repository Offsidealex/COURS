\newcommand{\capacites}{
  \begin{minipage}[c]{12.5cm}
    \begin{itemize}
      \item \rule{0ex}{2.5ex} Réaliser une solution de concentration en quantité de matière donnée par dilution ou dissolution. 
      \item  Calculer une masse molaire moléculaire à partir des masses molaires atomiques et de la formule chimique de la molécule.  
      \item Calculer la concentration en masse d’un soluté à partir de sa concentration en quantité de matière et de sa masse molaire moléculaire.  \rule[-1.5ex]{0ex}{3ex} 
      \end{itemize}
    \end{minipage}
  }




\newcommand{\connaissances}{
  \begin{minipage}[c]{12.5cm}
    \begin{itemize}
    \item \rule{0ex}{2.5ex}Connaître les définitions d'une solution, d'un solvant, d'un soluté.
    \item Connaître les relations entre la masse molaire $n$, la masse d’un échantillon $m$ et la quantité de matière $n$ 
    \item Connaître la relation entre la concentration en quantité de matière de soluté $c$, la quantité de matière $n$ et le volume de la solution $V$ 
    \item Connaître la déﬁnition de la concentration en masse d’un soluté dans une solution.  \rule[-1.5ex]{0ex}{3ex}
    \end{itemize}
    \end{minipage}
  }

\newcommand{\attitudes}{
  \begin{minipage}[c]{12.5cm}
    \begin{itemize}
    \item \rule{0ex}{2.5ex}Le sens de l'observation.
    \item L'ouverture à la communication, au dialogue et au débat argumenté.
    \item La rigueur et la précision
    \item Le respect des règles élémentaires de sécurité. 
    \item \rule[-1.5ex]{0ex}{3ex}Le gout de chercher et de raisonner.
    \end{itemize}
  \end{minipage}
}
