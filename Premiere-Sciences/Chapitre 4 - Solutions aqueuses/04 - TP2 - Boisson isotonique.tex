    \cadreblanc{15}
\vfill 

    \item En utilisant la formule du cours liant la concentration \textbf{c} en mol/L , le nombre de mole \textbf{n} et le volume \textbf{V}, $c=\dfrac{n}{V}$ , calculer , en mol, \textbf{n}, la quantité de matière de saccharose nécessaire à la préparation de la solution. 

    \cadreblanc{15}
    
    \vfill 
    
    \item Calculez la masse m, en g, de saccharose nécessaire à la préparation de cette solution, en utilisant la formule $n=\dfrac{m}{M}$. 
    
    \cadreblanc{15}

    \vfill

\fcolorbox{black}{blue!30}{\textbf{Protocole}}

\begin{itemize}[label=$\bullet$]
    \item Dans une coupelle, pesez 3,42g de sucre de saccharose en poudre.

    \vspace{3mm}

    \item Dans l’autre coupelle,pesez 0,12g de chlorure de sodium. 

    \vspace{3mm}

    \item Introduisez le sucre de saccharose puis le chlorure de sodium dans la fiole jaugée à l’aide de l’entonnoir.

    \vspace{3mm}

    \item Remplissez-la aux trois-quarts avec l’eau de la pissette.

    \vspace{3mm}
    
    \item Bouchez et agitez honrizontalement pour dissoudre. 

    \vspace{3mm}
    
    \item Ajoutez de l’eau jusqu’au trait de jauge. Rebouchez et agitez.
    
    
\end{itemize}
