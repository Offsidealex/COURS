\newcommand{\theme}{
Distinguer la puissance et l'énergie.}


\newcommand{\capacites}{
  \begin{minipage}[c]{12.5cm}
    \begin{itemize}
      \item \rule{0ex}{2.5ex}Mesurer la puissance électrique pendant une durée donnée par un appareil utilisé en régime continu. 
      \item  Calculer la puissance électrique et l'énergie électrique reçues pendant une durée donnée par un appareil en régime continu.  
      \item Établir expérimentalement la relation entre la puissance électrique reçue, la valeur de la tension et celle de l'intensité en régime continu. \rule[-1.5ex]{0ex}{3ex} 
      \end{itemize}
    \end{minipage}
  }


\newcommand{\connaissances}{
  \begin{minipage}[c]{12.5cm}
    \begin{itemize}
    \item \rule{0ex}{2.5ex}Connaitre la relation entre l'Energie électrique , la Puissance et la durée ($E=P \times t)$ 
    \item Connaître la relation entre la Puissance , la Tension et l'Intensité ($P=U \times I$)   
    \item Savoir que le Joule (J) est l'unité d'énergie du système international et connaître d'autres unités dont le kilowattheure (kWh)\rule[-1.5ex]{0ex}{3ex}
    \end{itemize}
    \end{minipage}
  }

\newcommand{\attitudes}{
  \begin{minipage}[c]{12.5cm}
    \begin{itemize}
    \item \rule{0ex}{2.5ex}Le sens de l'observation.
    \item L'ouverture à la communication, au dialogue et au débat argumenté.
    \item La rigueur et la précision
    \item Le respect des règles élémentaires de sécurité. 
    \item \rule[-1.5ex]{0ex}{3ex}Le gout de chercher et de raisonner.
    \end{itemize}
  \end{minipage}
}
