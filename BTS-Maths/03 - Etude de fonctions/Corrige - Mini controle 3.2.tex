\fcolorbox{black}{white}{\textbf{Exercice 2 : Savoir étudier la parité des fonctions}}\\[5pt]
Étudier la parité des fonctions suivantes :

\begin{multicols}{2}
\center{$f(x)=3x^2-1$}
\begin{tcolorbox}[colback=red!5!white,colframe=red!75!black]
Domaine de définition : $\mathbb{R}$, symétrique par rapport à 0.\\[3mm]
$f(-x) = 3(-x)^2-1 = 3x^2-1 = f(x)$\\[3mm]
\textbf{\textcolor{red}{$f$ est une fonction paire.}}
\end{tcolorbox}

\center{$g(x)=x^5+x^3+x$}
\begin{tcolorbox}[colback=red!5!white,colframe=red!75!black]
Domaine de définition : $\mathbb{R}$, symétrique par rapport à 0.\\[3mm]
$g(-x) = (-x)^5+(-x)^3+(-x) = -x^5-x^3-x = -g(x)$\\[3mm]
\textbf{\textcolor{red}{$g$ est une fonction impaire.}}
\end{tcolorbox}
\end{multicols}

\begin{multicols}{2}
\center{$h(x)=\dfrac{-5x}{2x^2+4}$}
\begin{tcolorbox}[colback=red!5!white,colframe=red!75!black]
Domaine : $\mathbb{R}$ (dénominateur toujours $> 0$), symétrique par rapport à 0.\\[3mm]
$h(-x) = \dfrac{-5(-x)}{2(-x)^2+4} = \dfrac{5x}{2x^2+4} = -h(x)$\\[3mm]
\textbf{\textcolor{red}{$h$ est une fonction impaire.}}
\end{tcolorbox}

\center{$j(x)=3x^2+\dfrac{1}{x^2}$}
\begin{tcolorbox}[colback=red!5!white,colframe=red!75!black]
Domaine : $\mathbb{R}^*$, symétrique par rapport à 0.\\[3mm]
$j(-x) = 3(-x)^2+\dfrac{1}{(-x)^2} = 3x^2+\dfrac{1}{x^2} = j(x)$\\[3mm]
\textbf{\textcolor{red}{$j$ est une fonction paire.}}
\end{tcolorbox}
\end{multicols}

\begin{multicols}{2}
\center{$k(x)=x^4-2x^2+5$}
\begin{tcolorbox}[colback=red!5!white,colframe=red!75!black]
Domaine de définition : $\mathbb{R}$, symétrique par rapport à 0.\\[3mm]
$k(-x) = (-x)^4-2(-x)^2+5 = x^4-2x^2+5 = k(x)$\\[3mm]
\textbf{\textcolor{red}{$k$ est une fonction paire.}}
\end{tcolorbox}

\center{$l(x)=\dfrac{x^3}{x^2+1}$}
\begin{tcolorbox}[colback=red!5!white,colframe=red!75!black]
Domaine : $\mathbb{R}$ (dénominateur toujours $> 0$), symétrique par rapport à 0.\\[3mm]
$l(-x) = \dfrac{(-x)^3}{(-x)^2+1} = \dfrac{-x^3}{x^2+1} = -l(x)$\\[3mm]
\textbf{\textcolor{red}{$l$ est une fonction impaire.}}
\end{tcolorbox}
\end{multicols}
